\section{Project Overview}

\begin{frame}{Project Goal}
\begin{block}{Objective}
Build a machine learning early-warning pipeline to monitor client business health and predict short-term risk and impact outcomes.
\end{block}

\begin{columns}[T]
\column{0.52\textwidth}
\textbf{Core ML Tracks}
\begin{itemize}
    \item Revenue prediction
    \item Employment prediction
    \item Risk scoring and tiering
\end{itemize}

\column{0.46\textwidth}
\textbf{Expected Value}
\begin{itemize}
    \item Earlier intervention signals
    \item Better portfolio monitoring
    \item Data-driven support prioritization
\end{itemize}
\end{columns}
\end{frame}

\section{Quality, Gaps, and Next Steps}

\begin{frame}{Current Strengths and Gaps}
\begin{columns}[T]
\column{0.49\textwidth}
\textbf{Strengths}
\begin{itemize}
    \item End-to-end ML flow exists
    \item Multi-target modeling implemented
    \item Saved models + predictions + metrics
    \item Track-specific documentation available
\end{itemize}

\column{0.49\textwidth}
\textbf{Gaps}
\begin{itemize}
    \item Centralized training pipeline/script not yet explicit
    \item Cross-model dashboarding not yet integrated
    \item Threshold calibration and drift checks pending
    \item Deployment packaging not yet shown in repo
\end{itemize}
\end{columns}
\end{frame}

\begin{frame}{Planned Milestones (Next Sprint)}
\begin{enumerate}
    \item Consolidate notebooks into reusable training/inference scripts.
    \item Standardize feature preprocessing and model versioning.
    \item Compare model families and tune top performers.
    \item Define early-warning thresholds and intervention rules.
    \item Build one reporting layer (CSV-to-dashboard/API-ready output).
    \item Add validation checks: data quality, leakage guardrails, drift baseline.
\end{enumerate}
\end{frame}

\begin{frame}{Requested Decisions / Support}
\begin{itemize}
    \item Confirm success criteria per track (revenue, employment, risk).
    \item Approve early-warning threshold framework.
    \item Confirm preferred deployment target (batch pipeline vs API).
    \item Validate priority user stories for intervention workflows.
\end{itemize}
\end{frame}
